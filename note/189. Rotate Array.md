> 开一个同样大小的数组，计算出最终结果的每一个数在原来结果中的位置

时间复杂度O(n), 空间复杂度O(n)

	class Solution {
	public:
	    void rotate(vector<int>& nums, int k) {
	        vector<int> ans;
	        int n = nums.size();
	        ans.assign(nums.begin(), nums.end());
	        for(int i = 0; i < n; i++) {
	            nums[(i + k) % n] = ans[i];
	        }
	        
	    }
	};
